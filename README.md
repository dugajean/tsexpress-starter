## Install

[npm][]:

```sh
npm install typescript-express
```

[yarn][]:

```sh
yarn add typescript-express
```


## Usage

```js
const TypescriptExpress = require('typescript-express');

const typescriptExpress = new TypescriptExpress();

console.log(typescriptExpress.renderName());
// script
```
